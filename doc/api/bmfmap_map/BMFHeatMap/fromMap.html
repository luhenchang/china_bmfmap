@override
dynamic fromMap(Map map) {
  if (null == map) {
    return null;
  }
  List data = map['data'] as List;
  return new BMFHeatMap(
      data: data
          ?.map((node) => BMFHeatMapNode.heatMapNode().fromMap(node))
          ?.toList(),
      radius: map['radius'],
      opacity: map['opacity'],
      gradient: BMFGradient.gradient().fromMap(map['gradient']));
}
