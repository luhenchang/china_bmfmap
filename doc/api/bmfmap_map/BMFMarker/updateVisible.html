Future<bool> updateVisible(bool visible) async {
  this.visible = visible;
  return await BMFMapDispatcherFactory.instance
      .getMarkerDispatcher()
      .updateMarkerMember(this.getMethodChannel(), {
    'id': this.getId(),
    'member': 'visible',
    'value': visible,
  });
}
