BMFInfoWindow.withMap(Map map) {
  if (null == map) {
    return;
  }

  this._id = map['id'];
  this.image = map['image'];
  this.coordinate = BMFCoordinate.coordinate().fromMap(map['coordinate']);
  this.yOffset = map['yOffset'];
  this.isAddWithBitmap = map['isAddWithBitmapDescriptor'];
}
